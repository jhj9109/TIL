
# Zip()

> 짝을 지을때 유용

```python
# zip()

justin_score = 0
neo_score = 0

for j, n in list(zip(justin,neo)):
    if j > n :
        justin_score+=1
    elif j < n :
        neo_score+=1
print(justin_score, neo_score)
```

```python
def is_contain_same_alphabet(a, b):
    # 아래에 코드를 작성하시오.
    result = False
    for k in a:
        for j in b:
            if k==j:

print(is_contain_same_alphabet('oreo', 'kikat'))
```
